const GOLF_API_KEY = "ERJJTZFDF7ZIKRUE76RRDOKLXQ";

const coursesToSearch = ["Deerhurst Highlands", "Muskoka Bay Club"];
let map;

async function fetchCourse(query) {
  const res = await fetch(`https://api.golfcourseapi.com/v1/search?search_query=${encodeURIComponent(query)}`, {
    headers: {
      Authorization: `Key ${GOLF_API_KEY}`,
    },
  });
  const data = await res.json();
  return data.courses?.[0]?.id || null;
}

async function getCourseDetails(courseId) {
  const res = await fetch(`https://api.golfcourseapi.com/v1/courses/${courseId}`, {
    headers: {
      Authorization: `Key ${GOLF_API_KEY}`,
    },
  });
  return await res.json();
}

async function initMap() {
  map = new google.maps.Map(document.getElementById("map"), {
    zoom: 13,
    center: { lat: 45.349, lng: -79.216 }, // Muskoka fallback center
  });

  for (const courseName of coursesToSearch) {
    const id = await fetchCourse(courseName);
    if (!id) {
      console.warn(`No course found for ${courseName}`);
      continue;
    }

    const courseData = await getCourseDetails(id);
    const { latitude, longitude } = courseData.location;

    new google.maps.Marker({
      position: { lat: latitude, lng: longitude },
      map,
      title: `${courseData.club_name} - ${courseData.course_name}`,
    });

    // Optional: zoom into each marker one by one (or cluster if needed)
  }
}

window.initMap = initMap;
